select e.empname, s.salary from employee e inner join salary s on e.empid = s.empid 
    where (e.empid, s.salary, e.department) in (select e1.empid, max(s1.salary), e1.department from employee e1 join salary s1 on e1.empid = s1.empid group by e1.department);
